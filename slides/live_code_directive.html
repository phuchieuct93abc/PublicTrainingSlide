Directive

1. Create directive with name "searchField" 

app.directive("searchField",function(){
    return {
        restrict : "E",
        scope:{value:"@"},
        templateUrl : "search-field.html",
        controller : function($scope){
            $scope.result = "Result:";
            $scope.onSubmit = function(){
             $scope.result = "Result:"+$scope.value;
            }
        }
    }
})
2. Create search-field.html template           

<div>
    <input type="text" ng-model="value"/>
    <button ng-click="onSubmit()">Search</button>
    <b ng-bind="result"></b>
</div>
3.Use directive <search-field> 

<search-field></search-field>
