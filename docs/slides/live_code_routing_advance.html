Routing - Advance

How to pass data from one view to others

1. Create a service to be able to share data between controllers. 

app.factory('myService', function() {
    var savedData = {}  
    return {
        set: function(data) {
             savedData = data;
        },
        get:  function () {
            return savedData;
        }
    }
});
2. In controller A 

app.controller("controllerA",function($scope,myService){
    myService.set("data")

})
2. In controller B 

app.controller("controllerB",function($scope,myService){
    $scope.savedData = myService.get()

})
