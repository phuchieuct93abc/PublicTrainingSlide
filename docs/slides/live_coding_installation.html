Installation

1. Download and include AngularJS in the HTML page

    <script src="angular.js"></script>
3. Pointing to AngularJS app via ng-app 

    index.html
    <html ng-app="myApp">


    app.js
    var app = angular.module("myApp",[])
4. Creating a view with ng-controller 

    index.html                     
    <body ng-controller="myController">
    </body>


    app.js
    app.controller("myController",function($scope){
        //Your code here    
    })
