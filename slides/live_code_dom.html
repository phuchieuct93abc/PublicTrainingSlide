Class manipulation by using ng-class

Requirement

Bootstrap class

How to do it


     /*Define class list in controller:*/
    $scope.classes = [
        {
            name:"Text danger",
            className:"text-danger"},
        {
            name:"Text success",
            className:"text-success"},
        {
            name:"Text primary",
            className:"text-primary"}
    ]


<div ng-class="myClass">Bootstrap class</div>  
<select ng-model="myClass" >
    <option ng-repeat="item in classes" ng-value="item.className">
        {{item.name}}
    </option>
</select>
